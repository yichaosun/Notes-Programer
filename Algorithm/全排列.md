// 1,2,3 => 全排列

class Solution{
  List<List<Integer>> result = new ArrayList<List<Integer>>();
  
  public List<List<Integer>> sovle(int[] candidates){
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,candidates);
    return result;
  }
  
  protected void dsf(List<Integer> combination,int[] candidates){
    if(combination.size() == candidates.length){
      result.add(new ArrayList<Integer>(combination));
      return;
    }
    for(int num : candidates){
      if(combination.contains(num)){
        continue;
      }
      combination.add(num);
      dsf(combination,candidates);
      combination.remove(combination.size()-1);
    }
  }
}
