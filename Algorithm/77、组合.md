// n,k
class Solution{
  private List<List<Integer>> result = new ArrayList<List<Integer>>();
  
  public List<List<Integer>> getCombination(int n, int k){
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,n,1,k);
    return result;
  }
  
  protected void dsf(List<Integer> combination,int n, int pos, int k){
    if(combination.size() == k){
      result.add(new ArrayList<Integer>(combination));
      return;
    }
    if(pos>n){
      return;
    }
    
    dsf(combination,n,pos+1,k);
    
    combination.add(pos);
    dsf(combination,n,pos,k);
    combination.remove(combination.size()-1);
  }
}
