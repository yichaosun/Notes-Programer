class Solution{
	private List<String> result = new ArrayList<String>();
  private String[] letterMap = String[]{
    "",
    "",
    "abc",
    "dfe",
    "ddd"
  };
  
  public List<String> getCombination(String digits){
    StringBuilder combination = StringBuilder();
    dsf(combination,0,digits);
    return result;
  }
  
  protected void dsf(StringBuilder combination,int pos,String digits){
    if(pos == digits.length()){
      result.add(combination.toString());
      return;
    }
    
    String options = letterMap[digits.charAt(pos)-'0'];
    if(options.isEmpty()){
      return;
    }
    
    for(int i=0;i<options.length;i++){
      combition.append(options.charAt(i));
      dsf(combination,pos+1,digits);
      combition.deleteCharAt(combition.length()-1);
    }
  }
}
