// subSet

class Solution{
  private List<List<Integer>> result = new ArrayList<List<Integer>>();
  
  public List<List<Integer>> getSubSet(int[] candidates){
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,candidates,0);
    return result;
  }
  
  protected void dsf(List<Integer> combination,int[] candidates,int pos){
    if(pos == candidates.length){
      result.add(new ArrayList<Integer>(combination));
      return;
    }
    dsf(combination,candidates,pos+1);
    combination.add(candidates[pos]);
		dsf(combination,candidates,pos+1);
    combination.remove(combination.size()-1);
  }
}
