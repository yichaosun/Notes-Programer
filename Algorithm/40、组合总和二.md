//1,1,3 candidates => combination

class Solution{
  private List<List<Integer>> result = new ArrayList<List<Integer>>();
  
  public List<List<Integer>> getCombination(int [] candidates, int target){
    List<int[]> freq = new ArrayList<int[]>();
    Arrays.sort(candidates);
    for(int num: candidates){
      int size = freq.size();
      if(freq.isEmpty() || freq.get(size-1)[0]!=num){
        freq.add(new int[]{num,1});
        continue;
      }
      ++freq.get(size-1)[1];
    }
    
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,freq,0,target);
    return result;
  }
  
  protected void dsf(List<Integer> combination,List<int[]> freq,int pos,int target){
    if(0 == target){
				result.add(new ArrayList<Integer>(combination));
      return;
    }
    if(pos == freq.size()){
      return;
    }
    dsf(combination,freq,pos+1,target);
    for(int i=1;i<= freq.get(pos)[1];i++){
      if(target>= i*freq.get(pos)[0]){
      	for(int j =1; j<=i;j++){
          combination.add(freq.get(pos)[0]);
        }  
        dsf(combination,freq,pos+1,target-i*freq.get(pos)[0]);
      	for(int j=1;j<=i;j++){
          combination.remove(combination.size()-1);
        }
      }
    }
  }
}
