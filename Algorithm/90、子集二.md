// 1,1,2 => subSet

class Solution{
  private List<List<Integer>> result = new ArrayList<Integer>();
  
  public List<List<Integer>> getSubSet(int[] candidates){
    Arrays.sort(candidates);
    List<int[]> freq = new ArrayList<int[]>();
    for(int num : candidates){
      int size = freq.size();
      if(0 == size || freq.get(size-1)[0]!=num){
        freq.add(new int[]{num,1});
        continue;
      }
      ++freq.get(size-1)[1];
    }
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,freq,0);
    return result;
  }
  
  protected void dsf(List<Integer> combination,List<int[]> freq,int pos){
    if(pos == freq.size()){
      result.add(new ArrayList<Integer>(combination));
      return;
    }
    dsf(combination,freq,pos+1);
    for(int i=1;i<=freq.get(pos)[1];i++){
      for(int j=1;j<=i;j++){
        combination.add(freq.get(pos)[0]);
      }
      dsf(combination,freq,pos+1);
      for(int j=1; j<=i;j++){
        combination.remove(combination.size()-1);
      }
    }
  }
}
