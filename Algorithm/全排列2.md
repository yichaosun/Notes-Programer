// 1,1,2=>全排列&不重复
class Solution{
  private List<List<Integer>> result = new ArrayList<List<Integer>>();
  
  public List<List<Integer>> solve(int[] candidates){
    Arrays.sort(candidates);
    boolean status[] = new boolean[candidates.length];
    Arrays.fill(status,false);
    
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,candidates,status);
    return result;
  }
  
  protected void dsf(List<Integer> combination,int[] candidates, boolean[] status){
    if(combination.size() == candidates.length){
      result.add(new ArrayList<Integer>(combination));
      return;
    }
    for(int i=0 ; i< candidates.length; i++){
      if(status[i] || (i>0 && candidates[i-1] == candidates[i] && !status[i-1]){
        continue;
      }
         status[i] = true;
         combination.add(candidates[i]);
         dsf(combination,candidates,status);
         status[i] = false;
         combination.remove(combination.size()-1);
    }
  }
}
