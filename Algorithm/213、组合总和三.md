// 9=>n,k
class Solution{
  private List<List<Integer>> result = new ArrayList<List<Integer>>();
  public List<List<Integer>> getCombination(int n, int k){
    List<Integer> combination = new ArrayList<Integer>();
    dsf(combination,9,1,n,k);
    return result;
  }
  
  protected void dsf(List<Integer> combination, int candidate, int pos, int target, int count){
    if(0 == target && combination.size() == count){
      result.add(new ArrayList<Integer>(combination));
      return;
    }
    if( pos>candidate || combination.size() == count){
      return;
    }
    dsf(combination,candidate,pos+1,target,count);
    
    if(target>=){
      combination.add(pos);
      dsf(combination,candidate,pos+1,target-pos,count);
      combination.remove(combination.size()-1);
    }
  }
}
