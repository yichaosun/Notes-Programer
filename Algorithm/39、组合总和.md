class Solution{
  private List<List<Integer>> result = new ArrayList<List<Integer>>();
  
  public List<List<Integer>> getCombination(int[] candidates,int target){
    	List<Integer> combination = new ArrayList<Integer>();
    	dsf(combination,candidates,0,target);
    	return result;
  }
  
  protected void dsf(List<Integer> combination,int[] candidates, int pos, int target){
    if(0 == target){
      combination.add(new ArrayList<Integer>(combination));
      return;
    }
    
    if(pos == candidates.length){
      return;
    }
    
    dsf(combination, candidates, pos+1, target);
    
    if(target >= candidates[pos]){
      combination.add(candidates[pos]);
      dsf(combination, candidates, pos, target-candidates[pos]);
      combination.remove(combination.size()-1);   
    }
  }
}
